#include<stdio.h>
int main()
{
	int a, b, s, h;
	scanf("--", &a, &b);
	s = a / b;
	scanf("%d", &h);
	printf("s = %d\n", s);
	printf("h = %d\n", h);
	system("pause");
	return 0;

}
//scanf中双引号部分是格式控制，要求在输入时以完全一致的格式才能准确输入数据
例如：
	scanf(" (%d, %d) ",&a, &b);
	输入时格式应为(2, 3) 这样a的值是2，b的值是3，
	括号和逗号的全角半角都得与scanf中的一致
文档：
	非空格字符，格式说明符（%）除外：不是空格字符（空白、换行字符、或
制表符）或格式说明符一部分（以%字符开头）的任何字符都会导致该函数读取下一个
字符从流中将其与该非空白字符进行比较，如果匹配，则将其丢弃，然后函数继续使用
format的下一个字符。如果字符不匹配，则函数将失败，返回，并保留流中的后续字符
未读。（后面再有scanf就会先尝试读取这部分数）

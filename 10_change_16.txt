#include<math.h>
int main()//十六进制转10进制
{
	char a[8] = { 0 };
	int i = 0;
	gets(a);
	int sum = 0;
	int tmp = 0;
	int sz = strlen(a);
	for (i = 0; i < sz; i++)
	{
		int k = a[i];
			if (k > 64)
			{
				tmp = (int)pow(16, sz - 1 - i);
				sum += (k - 55) * tmp;

			}
			else
			{
				tmp = (int)pow(16, sz - 1 - i);
				sum += (k - 48) * tmp;

			}
	}
	printf("%d", sum);

	system("pause");
	return 0;
}

#include<stdio.h>
#include<windows.h>
#pragma warning (disable:4996)

int main()
{
	char a[1000] = { 0 };
	char b[1000] = { 0 };
	gets(a);
	fflush(stdin);
	gets(b);
	int i = 0;
	int n = strlen(a);
	int j = 0;
	for (i = 0; i < n; i++)
	{
		if (a[i] == b[j])
			j++;
		else if (i != 0 && a[i - 1] == b[j])
		{
			while (i != 0 && a[i - 1] == b[j])
			{
				j++;
			}
		}
		else
		{
			printf("false");
			system("pause");
			return 0;
		}
	}

	printf("true");
	system("pause");
	return 0;
}
